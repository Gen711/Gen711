#!/usr/bin/make -rRsf 

MEM=5
CPU=5
RUN=run

all:/mnt/data3/lah/mattsclass/no_norm/testes.1.bless_corrected.fastq.gz /mnt/data3/lah/mattsclass/no_norm/testes.2.bless_corrected.fastq.gz /mnt/data3/lah/mattsclass/no_norm/testes.bless_no_norm_trinity.fasta 

#############################BLESS##############################################

/mnt/data3/lah/mattsclass/no_norm/testes.1.corrected.fastq.gz /mnt/data3/lah/mattsclass/no_norm/testes.2.corrected.fastq.gz:\ #output files#
/mnt/data3/lah/mattsclass/testes.R1.fastq /mnt/data3/lah/mattsclass/testes.R2.fastq #input files (raw reads)
        echo BEGIN ERROR CORRECTION: `date +'%a %d%b%Y  %H:%M:%S'`
        echo Results will be in a file named *corrected.fastq.gz
        echo Settings used: bless kmerlength = 25
        bless -kmerlength 25 -read1 /mnt/data3/lah/mattsclass/testes.R1.fastq -read2 /mnt/data3/lah/mattsclass/testes.R2.fastq -verify -notrim -prefix /mnt/data3/lah/mattsclass/no_norm/testes
        gzip /mnt/data3/lah/mattsclass/no_norm/testes.1.corrected.fastq /mnt/data3/lah/mattsclass/no_norm/testes.2.corrected.fastq &


#######################Trimmomatic/Trinity##########################

/mnt/data3/lah/mattsclass/testes.bless_no_norm_trinity.fasta:\ 
/mnt/data3/lah/mattsclass/testes.1.corrected.fastq.gz /mnt/data3/lah/mattsclass/testes.2.corrected.fastq.gz
	Trinity --seqType fq --JM 50G --trimmomatic --left /mnt/data3/lah/mattsclass/testes.1.corrected.fastq.gz --right /mnt/data3/lah/mattsclass/testes.2.corrected.fastq.gz --CPU 12 --output bless_no_norm_trinity.fasta \
	--quality_trimming_params "ILLUMINACLIP:/opt/trinity/trinity-plugins/Trimmomatic-0.30/adapters/TruSeq3-PE.fa:2:40:15 LEADING:2 TRAILING:2 MINLEN:25"
