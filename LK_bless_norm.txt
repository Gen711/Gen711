all:/home/lauren/Documents/p_eremicus/epi.1.corrected.fastq.gz /home/lauren/Documents/p_eremicus/epi.2.corrected.fastq.gz \
/home/lauren/Documents/p_eremicus/epi_corrected_internorm.fastq.gz \
/home/lauren/Documents/p_eremicus/epi_bless_norm_trinity.fasta

#############################BLESS##############################################

/home/lauren/Documents/p_eremicus/epi.1.corrected.fastq.gz /home/lauren/Documents/p_eremicus/epi.2.corrected.fastq.gz:\
/home/lauren/Documents/NYGenomeCenter/epidiymus.R1.fastq /home/lauren/Documents/NYGenomeCenter/epidiymus.R2.fastq
	echo BEGIN ERROR CORRECTION: `date +'%a %d%b%Y  %H:%M:%S'`
	echo Results will be in a file named *corrected.fastq.gz
	echo Settings used: bless kmerlength = 25
	bless -kmerlength 25 -read1 /home/lauren/Documents/NYGenomeCenter/epidiymus.R1.fastq -read2 /home/lauren/Documents/NYGenomeCenter/epidiymus.R2.fastq -verify -notrim -prefix epi
	gzip /home/lauren/Documents/p_eremicus/epi.1.corrected.fastq /home/lauren/Documents/p_eremicus/epi.2.corrected.fastq &


##########################khmer###############################

/home/lauren/Documents/p_eremicus/epi_corrected_internorm.fastq.gz:\
/home/lauren/Documents/p_eremicus/epi.1.corrected.fastq.gz /home/lauren/Documents/p_eremicus/epi.2.corrected.fastq.gz
	echo BEGIN NORMALIZATION `date +'%a %d%b%Y  %H:%M:%S'`
	echo Settings used: normalize-by-median.py -p -k 25 -C 50 -N 4 -x 15e9
	interleave-reads.py /home/lauren/Documents/p_eremicus/epi.1.corrected.fastq.gz /home/lauren/Documents/p_eremicus/epi.2.corrected.fastq.gz -o /home/lauren/Documents/p_eremicus/epi_corrected_internorm.fastq
	normalize-by-median.py -p -k 25  -C 50  -N 4 -x 15e9 --out  /home/lauren/Documents/p_eremicus/epi_corrected_internorm.fastq
	gzip epi_corrected_internorm.fastq &


#######################Trimmomatic/Trinity##########################

/home/lauren/Documents/p_eremicus/epi_bless_norm_trinity.fasta: \
/home/lauren/Documents/p_eremicus/epi_corrected_internorm.fastq.gz
	Trinity --seqType fq --JM 50G --trimmomatic --single $< --CPU 12 --output /home/lauren/Documents/p_eremicus/epi_bless_norm_trinity.fasta: \
	--quality_trimming_params "ILLUMINACLIP:/opt/trinityrnaseq-code/trinity-plugins/Trimmomatic/adapters/TruSeq3-PE.fa:2:40:15 LEADING:2 TRAILING:2 MINLEN:25"
